        experience_years: match.experience_years || 3,
        bio: match.bio
      }));
    } catch (error) {
      console.error('Database matching failed:', error);
      return getMockTrainers();
    }
  };

  // Mock data as last resort
  const getMockTrainers = (): Trainer[] => {
    const mockTrainers = [
      {
        trainer_id: 'mock-1',
        first_name: 'Alex',
        last_name: 'Johnson',
        headline: 'Certified Personal Trainer',
        specialties: ['Weight Loss', 'HIIT', 'Nutrition'],
        match_score: 92,
        experience_years: 5,
        bio: 'Specialized in weight loss transformations and nutritional guidance.',
        cluster: 'weight_loss_hiit'
      },
      {
        trainer_id: 'mock-2',
        first_name: 'Sarah',
        last_name: 'Miller',
        headline: 'Yoga & Mindfulness Coach',
        specialties: ['Yoga', 'Meditation', 'Flexibility'],
        match_score: 88,
        experience_years: 7,
        bio: 'Helping clients find balance through yoga and mindfulness practices.',
        cluster: 'yoga_mindfulness'
      }
    ];
    
    setDataSource('Mock Data (Fallback)');
    setUsingML(false);
    
    return mockTrainers;
  };

  // Log interaction events - TEMPORARY: Disabled without auth
  const logInteraction = async (trainerId: string, eventType: string) => {
    try {
      // TEMPORARY: Skip logging without proper auth
      console.log(`📊 [DEBUG] Would log ${eventType} event for trainer ${trainerId}`);
      return;
    } catch (error) {
      console.error('Failed to log interaction:', error);
    }
  };

  // Initialize user embedding
  const initializeUserEmbedding = async (userId: string, profile: any) => {
    try {
      console.log('🔄 Initializing embedding for user:', userId);
      const response = await fetch('/api/ml/embed/user', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          user_id: userId,
          profile_data: {
            goals: profile.fitness_goal ? [profile.fitness_goal] : [],
            fitness_level: profile.experience_level,
            workout_frequency: profile.workout_frequency,
            bio: profile.bio || 'Test user for ML system'
          }
        })
      });
      
      const data = await response.json();
      console.log('User embedding initialized:', data.success ? '✅ Success' : '❌ Failed', data);
      return data;
    } catch (error) {
      console.error('Failed to initialize user embedding:', error);
      return { success: false };
    }
  };

  // Load matches - TEMPORARY VERSION (bypasses auth)
  useEffect(() => {
    const loadMatches = async () => {
      setLoading(true);
      try {
        console.log('🚀 PHASE 14 ML TEST - Bypassing auth');
        console.log('Test User ID:', TEMP_USER_ID);
        
        // Create a mock profile for testing
        const mockProfile = {
          id: TEMP_USER_ID,
          fitness_goal: 'weight loss',
          experience_level: 'intermediate',
          workout_frequency: '3 times/week',
          bio: 'Test user for ML Phase 14 testing'
        };
        setUserProfile(mockProfile);
        
        // Initialize embedding for test user
        console.log('🔄 Generating user embedding...');
        const embeddingResult = await initializeUserEmbedding(TEMP_USER_ID, mockProfile);
        console.log('Embedding result:', embeddingResult);
        
        // Get the ACTUAL user ID that got the embedding
        const userIdForMatching = embeddingResult?.user_id || TEMP_USER_ID;
        setCurrentUserId(userIdForMatching);
        console.log('Using user ID for ML matching:', userIdForMatching);
        
        // Generate embeddings for some trainers if needed
        await generateTestTrainerEmbeddings();
        
        // Fetch matches based on ML toggle
        let matches: Trainer[];
        if (mlEnabled) {
          console.log('🤖 Using ML matching...');
          matches = await fetchMLMatches(userIdForMatching);
        } else {
          console.log('⚡ Using basic matching...');
          matches = await fetchDatabaseMatches();
        }
        
        console.log(`✅ Found ${matches.length} matches`);
        setTrainers(matches);
        
        // Log initial view events
        matches.forEach(match => {
          logInteraction(match.trainer_id, 'view');
        });
        
      } catch (error) {
        console.error('Error loading matches:', error);
        setTrainers(getMockTrainers());
      } finally {
        setLoading(false);
      }
    };

    loadMatches();
  }, [mlEnabled]);

  // Generate embeddings for test trainers
  const generateTestTrainerEmbeddings = async () => {
    try {
      console.log('🔄 Checking for trainers with embeddings...');
      const { data: trainers } = await supabase
        .from('trainer_profiles')
        .select('id, headline, bio, specialties, vector_embedding')
        .limit(5);
      
      if (!trainers || trainers.length === 0) {
        console.log('⚠️ No trainers found in database');
        return;
      }
      
      // Check which trainers need embeddings
      const trainersWithoutEmbeddings = trainers.filter(t => !t.vector_embedding);
      console.log(`Found ${trainers.length} trainers, ${trainersWithoutEmbeddings.length} need embeddings`);
      
      // Generate embeddings for first 2 trainers if needed
      for (const trainer of trainersWithoutEmbeddings.slice(0, 2)) {
        console.log(`🔄 Generating embedding for trainer: ${trainer.id}`);
        try {
          await fetch('/api/ml/embed/trainer', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
              trainer_id: trainer.id,
              trainer_data: {
                bio: trainer.bio || 'Certified personal trainer',
                specialties: trainer.specialties || ['General Fitness'],
                experience_years: 3,
                headline: trainer.headline || 'Personal Trainer'
              }
            })
          });
          console.log(`✅ Embedding generated for trainer ${trainer.id}`);
        } catch (error) {
          console.log(`⚠️ Skipping embedding for ${trainer.id}:`, error.message);
        }
      }
      
    } catch (error) {
      console.error('Failed to generate trainer embeddings:', error);
    }
  };

  // TEMPORARY: Simplified handlers without auth
  const handleLike = (trainerId: string) => {
    console.log(`👍 Liked trainer ${trainerId}`);
    logInteraction(trainerId, 'like');
    alert('Like functionality (demo) - Would save trainer');
  };

  const handleBook = (trainerId: string) => {
    console.log(`📅 Booking trainer ${trainerId}`);
    logInteraction(trainerId, 'book');
    alert('Booking functionality (demo) - Would book session');
  };

  // NEW: Message handler
  const handleMessage = (trainerId: string) => {
    console.log(`💬 Messaging trainer ${trainerId}`);
    logInteraction(trainerId, 'message_click');
    router.push(`/messages/${trainerId}`);
  };

  const toggleML = () => {
    setMlEnabled(!mlEnabled);
    setShowDebug(false);
    console.log(`ML ${!mlEnabled ? 'enabled' : 'disabled'}`);
  };

  const getMatchColor = (score: number) => {
    if (score >= 90) return 'from-green-500 to-emerald-600';
    if (score >= 80) return 'from-blue-500 to-cyan-600';
    if (score >= 70) return 'from-yellow-500 to-amber-600';
    return 'from-gray-500 to-slate-600';
  };

  const getSpecialtyIcon = (specialty: string) => {
    const lower = specialty.toLowerCase();
    if (lower.includes('yoga') || lower.includes('meditation')) return '🧘';
    if (lower.includes('hiit') || lower.includes('cardio')) return '⚡';
    if (lower.includes('strength') || lower.includes('weight')) return '🏋️';
    if (lower.includes('nutrition')) return '🥗';
    if (lower.includes('beginner')) return '🌱';
    if (lower.includes('rehab')) return '🏥';
    return '💪';
  };

  // Skeleton loader
  if (loading) {
    return (
      <div className="min-h-screen bg-gradient-to-br from-gray-900 via-black to-gray-900 p-4 md:p-8">
        <div className="max-w-7xl mx-auto">
          <div className="flex flex-col md:flex-row justify-between items-start md:items-center mb-8">
            <div>
              <div className="h-10 w-64 bg-gray-800/50 rounded mb-2 animate-pulse"></div>
              <div className="h-6 w-48 bg-gray-800/50 rounded animate-pulse"></div>
            </div>
            <div className="h-10 w-48 bg-gray-800/50 rounded mt-4 md:mt-0 animate-pulse"></div>
          </div>
          
          <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
            {[1, 2, 3, 4, 5, 6].map((i) => (
              <div key={i} className="h-[400px] w-full bg-gray-800/50 rounded-xl animate-pulse"></div>
            ))}
          </div>
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-gradient-to-br from-gray-900 via-black to-gray-900 p-4 md:p-8">
      <div className="max-w-7xl mx-auto">
        {/* Header */}
        <div className="flex flex-col md:flex-row justify-between items-start md:items-center mb-8">
          <div>
            <h1 className="text-3xl md:text-4xl font-bold text-white mb-2">
              Your Perfect Matches
              {usingML && (
                <span className="ml-3 inline-flex items-center px-3 py-1 rounded-full text-sm font-medium bg-gradient-to-r from-purple-900/50 to-blue-900/50 text-blue-200 border border-blue-700/50">
                  <span className="mr-2">🤖</span>
                  AI-Powered
                </span>
              )}
            </h1>
            <p className="text-gray-400">
              {usingML 
                ? 'Matches powered by machine learning algorithms'
                : 'Matches based on your profile preferences'}
              <span className="ml-2 text-sm text-gray-500">({dataSource})</span>
            </p>
            <div className="mt-2 text-xs text-yellow-400">
              🔧 Phase 14 ML Test Mode | User: {currentUserId.substring(0, 15)}...
            </div>
          </div>
          
          <div className="flex flex-col sm:flex-row gap-3 mt-4 md:mt-0">
            <button
              onClick={toggleML}
              className={`flex items-center px-4 py-2 rounded-lg font-medium transition-all ${
                mlEnabled 
                  ? 'bg-gradient-to-r from-purple-600 to-blue-600 text-white hover:from-purple-700 hover:to-blue-700' 
                  : 'border border-gray-700 text-gray-300 hover:bg-gray-800'
              }`}
            >
              {mlEnabled ? (
                <>
                  <span className="mr-2">🤖</span>
                  AI Matching ON
                </>
              ) : (
                <>
                  <span className="mr-2">⚡</span>
                  AI Matching OFF
                </>
              )}
            </button>
            
            {/* NEW: Messages Button */}
            <Link 
              href="/messages"
              className="flex items-center px-4 py-2 rounded-lg font-medium border border-gray-700 text-gray-300 hover:bg-gray-800 transition-all"
            >
              <span className="mr-2">💬</span>
              Messages
            </Link>
            
            <button
              onClick={() => setShowDebug(!showDebug)}
              className="flex items-center px-4 py-2 rounded-lg font-medium border border-gray-700 text-gray-300 hover:bg-gray-800 transition-all"
            >
              <span className="mr-2">🔧</span>
              {showDebug ? 'Hide Debug' : 'Show Debug'}
            </button>
          </div>
        </div>

        {/* Debug Panel */}
        {showDebug && (
          <div
            className="mb-8 p-6 bg-gray-900/80 backdrop-blur-sm border border-gray-700 rounded-xl"
          >
            <div className="flex items-center justify-between mb-4">
              <h3 className="text-lg font-semibold text-white flex items-center">
                <span className="mr-2">📊</span>
                System Diagnostics
              </h3>
              <span className={`px-3 py-1 rounded-full text-sm font-medium ${
                usingML 
                  ? 'bg-green-900/50 text-green-200' 
                  : 'bg-gray-800 text-gray-300'
              }`}>
                {usingML ? 'ML ACTIVE' : 'ML INACTIVE'}
              </span>
            </div>
            
            <div className="grid grid-cols-1 md:grid-cols-3 gap-4 mb-4">
              <div className="p-3 bg-gray-800/50 rounded-lg">
                <div className="text-sm text-gray-400">Data Source</div>
                <div className="font-medium text-white">{dataSource}</div>
              </div>
              <div className="p-3 bg-gray-800/50 rounded-lg">
                <div className="text-sm text-gray-400">Trainers Found</div>
                <div className="font-medium text-white">{trainers.length}</div>
              </div>
              <div className="p-3 bg-gray-800/50 rounded-lg">
                <div className="text-sm text-gray-400">Average Match Score</div>
                <div className="font-medium text-white">
                  {trainers.length > 0 
                    ? Math.round(trainers.reduce((acc, t) => acc + t.match_score, 0) / trainers.length)
                    : 0}%
                </div>
              </div>
            </div>
            
            <div className="text-sm text-gray-400 space-y-1">
              <div className="flex items-center">
                <div className={`w-2 h-2 rounded-full mr-2 ${mlEnabled ? 'bg-green-500' : 'bg-yellow-500'}`}></div>
                <span>ML Engine: {mlEnabled ? 'Enabled' : 'Disabled'}</span>
              </div>
              <div className="flex items-center">
                <div className="w-2 h-2 rounded-full mr-2 bg-blue-500"></div>
                <span>Test Mode: Active (bypassing auth)</span>
              </div>
              <div className="flex items-center">
                <div className="w-2 h-2 rounded-full mr-2 bg-purple-500"></div>
                <span>Weighted Scoring: 0.55·similarity + 0.25·cluster + 0.20·behavioral</span>
              </div>
              <div className="flex items-center">
                <div className="w-2 h-2 rounded-full mr-2 bg-orange-500"></div>
                <span>User ID: {currentUserId.substring(0, 30)}...</span>
              </div>
              <div className="flex items-center">
                <div className="w-2 h-2 rounded-full mr-2 bg-green-400"></div>
                <span>Messaging: Integrated (Phase 15 Complete)</span>
              </div>
            </div>
          </div>
        )}

        {/* User Profile Summary */}
        {userProfile && (
          <div 
            className="mb-8 p-6 bg-gradient-to-r from-gray-800/40 to-gray-900/40 backdrop-blur-sm border border-gray-700 rounded-xl"
          >
            <div className="flex items-center justify-between">
              <div className="flex items-center">
                <div className="p-3 bg-gradient-to-br from-blue-900/30 to-purple-900/30 rounded-lg mr-4">
                  <span className="text-blue-300">👤</span>
                </div>
                <div>
                  <h3 className="text-lg font-semibold text-white">Test Profile (Phase 14 Demo)</h3>
                  <div className="flex flex-wrap gap-2 mt-2">
                    {userProfile.fitness_goal && (
                      <span className="px-3 py-1 rounded-full text-sm bg-blue-900/50 text-blue-200 border border-blue-700/50 flex items-center">
                        <span className="mr-1">🎯</span>
                        {userProfile.fitness_goal}
                      </span>
                    )}
                    {userProfile.experience_level && (
                      <span className="px-3 py-1 rounded-full text-sm bg-purple-900/50 text-purple-200 border border-purple-700/50 flex items-center">
                        <span className="mr-1">🏆</span>
                        {userProfile.experience_level}
                      </span>
                    )}
                    {userProfile.workout_frequency && (
                      <span className="px-3 py-1 rounded-full text-sm bg-green-900/50 text-green-200 border border-green-700/50 flex items-center">
                        <span className="mr-1">🔥</span>
                        {userProfile.workout_frequency}
                      </span>
                    )}
                  </div>
                </div>
              </div>
              
              {usingML && (
                <div className="hidden md:block text-right">
                  <div className="text-sm text-gray-400">Matching Method</div>
                  <div className="flex items-center text-green-400">
                    <span className="mr-2">🤖</span>
                    <span className="font-medium">AI-Powered</span>
                  </div>
                </div>
              )}
            </div>
          </div>
        )}

        {/* Tab Navigation */}
        <div className="flex space-x-1 mb-8 p-1 bg-gray-900/50 rounded-lg w-fit">
          <button
            onClick={() => setActiveTab('all')}
            className={`px-4 py-2 rounded-md text-sm font-medium transition-all ${activeTab === 'all' 
              ? 'bg-gradient-to-r from-blue-600 to-purple-600 text-white shadow-lg' 
              : 'text-gray-400 hover:text-white hover:bg-gray-800/50'}`}
          >
            All Matches ({trainers.length})
          </button>
          <button
            onClick={() => setActiveTab('top')}
            className={`px-4 py-2 rounded-md text-sm font-medium transition-all ${activeTab === 'top' 
              ? 'bg-gradient-to-r from-blue-600 to-purple-600 text-white shadow-lg' 
              : 'text-gray-400 hover:text-white hover:bg-gray-800/50'}`}
          >
            Top Picks ({Math.min(3, trainers.length)})
          </button>
        </div>

        {/* Matches Grid */}
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
          {trainers
            .filter(trainer => activeTab === 'all' || trainer.match_score >= 85)
            .slice(0, activeTab === 'top' ? 3 : trainers.length)
            .map((trainer, index) => (
              <div
                key={trainer.trainer_id}
                className="h-full overflow-hidden border border-gray-800 bg-gradient-to-b from-gray-900/80 to-black/80 backdrop-blur-sm hover:border-gray-700 transition-all duration-300 rounded-xl"
                onMouseEnter={() => logInteraction(trainer.trainer_id, 'view')}
              >
                <div className="p-6 relative">
                  {/* Match Score Badge */}
                  <div className="absolute top-4 right-4">
                    <div className={`px-3 py-1.5 rounded-full bg-gradient-to-r ${getMatchColor(trainer.match_score)} text-white font-bold text-sm shadow-lg`}>
                      {trainer.match_score}% Match
                    </div>
                  </div>
                  
                  {/* Trainer Header */}
                  <div className="flex items-start mb-6">
                    <div className="flex-shrink-0 mr-4">
                      <div className="w-16 h-16 rounded-full bg-gradient-to-br from-blue-900/30 to-purple-900/30 flex items-center justify-center text-2xl text-white">
                        {trainer.first_name.charAt(0)}
                      </div>
                    </div>
                    <div>
                      <h3 className="text-xl font-bold text-white">
                        {trainer.first_name} {trainer.last_name}
                      </h3>
                      <p className="text-gray-400 text-sm mt-1">{trainer.headline}</p>
                      <div className="flex items-center mt-2">
                        <span className="mr-1">⭐</span>
                        <span className="text-sm text-gray-300">
                          {trainer.experience_years}+ years experience
                        </span>
                      </div>
                    </div>
                  </div>
                  
                  {/* Specialties */}
                  <div className="mb-6">
                    <h4 className="text-sm font-medium text-gray-400 mb-3">Specialties</h4>
                    <div className="flex flex-wrap gap-2">
                      {trainer.specialties.slice(0, 4).map((specialty, idx) => (
                        <span 
                          key={idx} 
                          className="px-3 py-1 rounded-full text-sm bg-gray-800/50 text-gray-300 border border-gray-700"
                        >
                          {getSpecialtyIcon(specialty)} {specialty}
                        </span>
                      ))}
                      {trainer.specialties.length > 4 && (
                        <span className="px-3 py-1 rounded-full text-sm bg-gray-800/50 text-gray-400 border border-gray-700">
                          +{trainer.specialties.length - 4} more
                        </span>
                      )}
                    </div>
                  </div>
                  
                  {/* Bio */}
                  {trainer.bio && (
                    <div className="mb-6">
                      <p className="text-gray-300 text-sm line-clamp-3">{trainer.bio}</p>
                    </div>
                  )}
                  
                  {/* Match Progress Bar */}
                  <div className="mb-6">
                    <div className="flex justify-between text-sm mb-2">
                      <span className="text-gray-400">Compatibility</span>
                      <span className="font-medium text-white">{trainer.match_score}%</span>
                    </div>
                    <div className="h-2 bg-gray-800 rounded-full overflow-hidden">
                      <div 
                        className={`h-full bg-gradient-to-r ${getMatchColor(trainer.match_score)}`}
                        style={{ width: `${trainer.match_score}%` }}
                      ></div>
                    </div>
                  </div>
                  
                  {/* ML Debug Info */}
                  {showDebug && trainer.cluster && (
                    <div className="mb-6 p-3 bg-gray-900/50 rounded-lg">
                      <div className="text-xs text-gray-400 mb-1">ML Details</div>
                      <div className="grid grid-cols-2 gap-2 text-sm">
                        <div>
                          <span className="text-gray-500">Cluster:</span>
                          <span className="ml-2 text-blue-300">{trainer.cluster}</span>
                        </div>
                        {trainer.similarity_score && (
                          <div>
                            <span className="text-gray-500">Similarity:</span>
                            <span className="ml-2 text-green-300">
                              {(trainer.similarity_score * 100).toFixed(1)}%
                            </span>
                          </div>
                        )}
                      </div>
                    </div>
                  )}
                  
                  {/* Action Buttons - UPDATED WITH MESSAGING */}
                  <div className="grid grid-cols-2 gap-3">
                    <button
                      onClick={() => handleLike(trainer.trainer_id)}
                      className="border border-gray-700 text-gray-300 hover:bg-gray-800 hover:text-white hover:border-gray-600 rounded-lg py-2 px-3 flex items-center justify-center transition-all text-sm"
                    >
                      <span className="mr-1">⭐</span>
                      Save
                    </button>
                    
                    {/* NEW: Message Button */}
                    <button
                      onClick={() => handleMessage(trainer.trainer_id)}
                      className="border border-blue-700 text-blue-300 hover:bg-blue-900/30 hover:text-white hover:border-blue-600 rounded-lg py-2 px-3 flex items-center justify-center transition-all text-sm"
                    >
                      <span className="mr-1">💬</span>
                      Message
                    </button>
                    
                    <button
                      onClick={() => handleBook(trainer.trainer_id)}
                      className="col-span-2 bg-gradient-to-r from-blue-600 to-purple-600 hover:from-blue-700 hover:to-purple-700 text-white rounded-lg py-2 px-4 transition-all mt-1"
                    >
                      Book Session
                    </button>
                  </div>
                </div>
              </div>
            ))}
        </div>

        {/* Empty State */}
        {trainers.length === 0 && !loading && (
          <div className="text-center py-16">
            <div className="w-24 h-24 mx-auto mb-6 rounded-full bg-gradient-to-br from-gray-800 to-gray-900 flex items-center justify-center">
              <span className="text-gray-600 text-2xl">👤</span>
            </div>
            <h3 className="text-xl font-semibold text-white mb-2">No Matches Found</h3>
            <p className="text-gray-400 max-w-md mx-auto mb-8">
              We couldn't find any trainers matching your preferences. Try adjusting your profile settings or check back later.
            </p>
            <button
              onClick={() => router.push('/profile')}
              className="border border-gray-700 text-gray-300 hover:bg-gray-800 rounded-lg py-2 px-4 transition-all"
            >
              Edit Profile
            </button>
          </div>
        )}

        {/* Footer Note */}
        <div className="mt-12 pt-8 border-t border-gray-800/50">
          <div className="text-center text-gray-500 text-sm">
            <p className="mb-2">
              {usingML ? (
                <>
                  <span className="mr-1">🤖</span>
                  Matches are calculated using machine learning algorithms that consider your profile, preferences, and behavior.
                </>
              ) : (
                'Matches are based on your profile preferences and trainer specialties.'
              )}
            </p>
            <p>
              Found {trainers.length} trainer{trainers.length !== 1 ? 's' : ''} • 
              Last updated just now • 
              <button 
                onClick={toggleML}
                className="ml-1 text-blue-400 hover:text-blue-300 underline"
              >
                {mlEnabled ? 'Switch to basic matching' : 'Enable AI matching'}
              </button>
            </p>
            <div className="mt-2 text-xs text-yellow-600">
              🔧 Phase 14 ML Test Mode Active | This is a demonstration of the ML matching system
            </div>
            <div className="mt-1 text-xs text-green-500">
              💬 Phase 15 Complete: Messaging integrated with ML matches
            </div>
          </div>
        </div>
      </div>
    </div>
  );
}
